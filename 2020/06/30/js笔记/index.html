continue的使用

应用在循环中，指的是跳过本次循环，进行下一次循环

    function add(n){
        var sum = 0;
        for(var i=1;i<=n;i++){
         if(i%2==0){
            continue;  
            }
        sum+=i;
        }
        return sum;
    }
console.log(add(10));

typeof操作符和显式类型转化

六种数据类型：number,string, boolean,undefind,object,function.
typeof()用法：
    区分123和"123"
    var num = 123;
    console.log(typeof(num));//number
    var num = "123";
    console.log(typeof(num));//string
    var obj ={};   
    console.log(typeof(obj)); //object
    var arr =[];
    console.log(typeof(arr)); //object
    var bol = true;
    console.log(typeof(bol));  //boolean
    ar func = function () {}
    console.log(typeof(func));  //function
    var und = undefined;
    console.log(typeof(und));  //undefind

类型转换

var num = Number("123");//引号内一定要是数字，否则类型为NaN
console.log(typeof(num)+":"+num);//number：123
var demo = "123abc";

var num = parseInt(demo);
console.log(typeof(num)+":"+num);//number:123

var demo1 = "10"
var num1 = parseInt(demo1,16);//demo1为16进制，将16进制转为十进制后赋值给num1
console.log(typeof(num1)+":"+num1);

var demo2 = "true";
var num2 = Boolean(demo2);
console.log(typeof(num2)+":"+num2);//boolean:true

var demo3 = undefind;
var num3 = String(demo3);
console.log(typeof(num3)+":"+num3);//string:undefind
